building [icon: building, color: blue] {
  id serial pk
  name VARCHAR(50) not null
  createdAt timestamp
  updatedAt timestamp
}

floor [icon: rectangle-horizontal, color: green] {
  id serial pk
  number int not null
  building_id int fk
  createdAt timestamp
  updatedAt timestamp
}

elevator [icon: rectangle-vertical, color: orange] {
  id serial pk
  building_id int fk
  elevator_shaft_id int fk unique
  name VARCHAR(20) not null
  status enum[idle, moving, maintenance] 
  createdAt timestamp
  updatedAt timestamp
}

floor_request [icon: git-pull-request, color: purple] {
  id serial pk
  from_floor_id int fk
  to_floor_id int fk
  status enum[pending, assigned, completed]
  createdAt timestamp
  updatedAt timestamp
}

elevator_shaft [icon: triangle, color: red] {
  id serial pk
  building_id int fk
  createdAt timestamp
  updatedAt timestamp
}

ride_assignment [icon: ruler, color: yellow] {
  id serial pk
  floor_request_id int fk
  elevator_id int fk
  status enum[pending, completed, failed]
  createdAt timestamp
  updatedAt timestamp
}

ride_log [icon: log, color: red] {
  id serial pk
  ride_assignment_id int fk 
  startedAt timestamp
  completedAt timestamp
  createdAt timestamp
  updatedAt timestamp
}

elevator_floor [icon: sum, color: red] {
  elevator_id int fk
  floor_id int fk
  createdAt timestamp
  updatedAt timestamp
}

maintenance [icon: mainframe, color: purple] {
  id serial pk
  elevator_id int fk
  description TEXT
  startedAt timestamp
  completedAt timestamp
  status enum[completed, pending, failed]
  createdAt timestamp
  updatedAt timestamp
}

elevator_floor.floor_id > floor.id

elevator_shaft.id - elevator.elevator_shaft_id

ride_assignment.id - ride_log.ride_assignment_id
elevator.id < maintenance.elevator_id
elevator.id < ride_assignment.elevator_id
